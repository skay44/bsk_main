64def decrypt_rsa_key(encrypted_rsa_key, pin, salt, iv):
65 kdf = PBKDF2HMAC(
66 algorithm=hashes.SHA256(),
67 length=32,
68 salt=salt,
69 iterations=100000,
70 )
71 aes_key = kdf.derive(pin.encode())
72
73 cipher = Cipher(algorithms.AES(aes_key), modes.CFB(iv))
74 decryptor = cipher.decryptor()
75 decrypted = decryptor.update(encrypted_rsa_key) + decryptor.finalize()
76
77 try:
78 priv_key = serialization.load_pem_private_key(decrypted, password=None)
79 except ValueError as e:
80 raise ValueError("Nieprawidłowy PIN lub uszkodzony plik RSA") from e
81
82 return priv_key

88def sign_pfd(pdf_path, priv_key, output_path):
89 # Read PDF bytes
90 with open(pdf_path, "rb") as f:
91 pdf_bytes = f.read()
92
93 digest = hashlib.sha256(pdf_bytes).digest()
94
95 signature = priv_key.sign(
96 digest,
97 padding=padding.PKCS1v15(),
98 algorithm=hashes.SHA256()
99 )
100
101 with open(output_path, "wb") as f_out:
102 f_out.write(pdf_bytes)
103 f_out.write(b"\n%%SIGNATURE%%\n")
104 f_out.write(signature)

118def validate_pdf(pdf_path, public_key):
119 with open(pdf_path, "rb") as f:
120 content = f.read()
121
122 if b"\n%%SIGNATURE%%\n" not in content:
123 raise ValueError("PDF nie jest podpisany")
124
125 pdf_bytes, signature = content.split(b"\n%%SIGNATURE%%\n")
126 digest = hashlib.sha256(pdf_bytes).digest()
127
128 try:
129 public_key.verify(
130 signature,
131 digest,
132 padding.PKCS1v15(),
133 hashes.SHA256()
134 )
135 return True
136 except Exception:
137 return False

191 def execute_app(self):
192 self.pendrive_path = find_pendrive()
193 if self.pendrive_path:
194 messagebox.showinfo("Znaleziono pendrive: ", f"Pendrive found at {self.pendrive_path}")
195
196 public_key = read_public_key(self.pendrive_path[0])
197 salt, iv, encrypted_key = read_encrypted_rsa_key(self.pendrive_path[0])
198 pin = self.pin_entry.get()
199 try:
200 priv_key = decrypt_rsa_key(encrypted_key, pin, salt, iv)
201 except ValueError as e:
202 messagebox.showerror("Błąd", "Niepoprawny klucz prywatny lub niepoprawny pin")
203 return
204
205 if (self.mode.get() == "Validate"):
206 try:
207 result = validate_pdf(self.selected_pdf, public_key)
208 except ValueError as e:
209 messagebox.showerror("Błąd", "PDF nie jest podpisany")
210 return
211 if (result):
212 messagebox.showinfo("Sukces!", "PDF podpisany poprawnie!")
213 else:
214 messagebox.showinfo("Błąd!", "PDF podpisany niepoprawnie lub zmodyfikowany!")
215 if (self.mode.get() == "Sign"):
216 sign_pfd(self.selected_pdf, priv_key, signed_output_path(self.selected_pdf))
